local function Tween(obj, props, duration, style, direction)
    style     = style     or Enum.EasingStyle.Quart
    direction = direction or Enum.EasingDirection.Out
    local t = TweenService:Create(obj, TweenInfo.new(duration or 0.25, style, direction), props)
    t:Play()
    return t
end

local function SafeCall(f, ...)
    if type(f) == "function" then
        local ok, err = pcall(f, ...)
        if not ok then warn("[Mixusfield] Callback error: " .. tostring(err)) end
    end
end

local function New(class, props)
    local obj = Instance.new(class)
    for k, v in pairs(props or {}) do
        if k ~= "Parent" then pcall(function() obj[k] = v end) end
    end
    if props and props.Parent then obj.Parent = props.Parent end
    return obj
end

-- Tag: ThemeRole + which colour property to animate
local function Tag(obj, role, colorProp)
    pcall(function()
        obj:SetAttribute("ThemeRole", role)
        obj:SetAttribute("ThemeColorProp", colorProp or "BackgroundColor3")
    end)
    return obj
end

local function MakePadding(parent, t, b, l, r)
    return New("UIPadding", {
        Parent = parent,
        PaddingTop    = UDim.new(0, t or 0),
        PaddingBottom = UDim.new(0, b or 0),
        PaddingLeft   = UDim.new(0, l or 0),
        PaddingRight  = UDim.new(0, r or 0),
    })
end

local function MakeList(parent, padding, dir)
    return New("UIListLayout", {
        Parent              = parent,
        Padding             = UDim.new(0, padding or 6),
        FillDirection       = dir or Enum.FillDirection.Vertical,
        HorizontalAlignment = Enum.HorizontalAlignment.Center,
        SortOrder           = Enum.SortOrder.LayoutOrder,
    })
end

local function MakeCorner(parent, radius)
    return New("UICorner", { Parent = parent, CornerRadius = UDim.new(0, radius or 10) })
end

local function MakeStroke(parent, color, thickness, transparency)
    return New("UIStroke", {
        Parent       = parent,
        Color        = color or Color3.fromRGB(60, 20, 20),
        Thickness    = thickness or 1,
        Transparency = transparency or 0,
    })
end

local function IsMobile()
    return UserInputService.TouchEnabled and not UserInputService.KeyboardEnabled
end

local function ScreenSize() return Camera.ViewportSize end

local function NormaliseAsset(str)
    if type(str) ~= "string" then return nil end
    if str:match("^%d+$")         then return "rbxassetid://" .. str end
    if str:match("^rbxassetid://") then return str end
    return nil
end

-- SAVE SYSTEM
-- ════════════════════════════════════════════════════════════════
local SaveSystem = {}
SaveSystem.__index = SaveSystem

function SaveSystem.new(configName)
    local self = setmetatable({}, SaveSystem)
    self.ConfigName = configName or "MixusfieldConfig"
    self.Data = {}
    self:Load()
    return self
end
function SaveSystem:Load()
    pcall(function()
        if not isfolder then return end
        if not isfolder("Mixusfield") then makefolder("Mixusfield") end
        local path = "Mixusfield/" .. self.ConfigName .. ".json"
        if isfile and isfile(path) then
            self.Data = HttpService:JSONDecode(readfile(path)) or {}
        end
    end)
end
function SaveSystem:Save()
    pcall(function()
        if not writefile then return end
        if makefolder and not isfolder("Mixusfield") then makefolder("Mixusfield") end
        writefile("Mixusfield/" .. self.ConfigName .. ".json", HttpService:JSONEncode(self.Data))
    end)
end
function SaveSystem:Set(key, value) self.Data[key] = value; self:Save() end
function SaveSystem:Get(key, default)
    if self.Data[key] ~= nil then return self.Data[key] end
    return default
end

local NotifActive = 0
local MAX_NOTIFS  = 4

local function ShowNotification(options, theme, screenGui)
    if NotifActive >= MAX_NOTIFS then return end
    NotifActive += 1

    local titleTxt  = options.Title    or "Notification"
    local bodyTxt   = options.Content  or ""
    local dur       = options.Duration or 3
    local nType     = options.Type     or "Info"
    local accentCol = ({ Info=theme.Accent, Success=Color3.fromRGB(30,200,120),
                         Warning=Color3.fromRGB(255,180,30), Error=Color3.fromRGB(255,60,60) })[nType] or theme.Accent

    local holder = screenGui:FindFirstChild("MixusNotifHolder")
    if not holder then
        holder = New("Frame", {
            Name = "MixusNotifHolder", Parent = screenGui,
            BackgroundTransparency = 1,
            AnchorPoint = Vector2.new(1,1),
            Position    = UDim2.new(1,-16,1,-16),
            Size        = UDim2.new(0,300,1,-16),
            ZIndex      = 50,
        })
        New("UIListLayout", { Parent=holder, Padding=UDim.new(0,8),
            FillDirection=Enum.FillDirection.Vertical,
            VerticalAlignment=Enum.VerticalAlignment.Bottom,
            SortOrder=Enum.SortOrder.LayoutOrder })
    end

    local Notif = New("Frame", {
        Parent=holder, BackgroundColor3=theme.Card,
        Size=UDim2.new(1,0,0,0), AutomaticSize=Enum.AutomaticSize.Y,
        BackgroundTransparency=0.04, ClipsDescendants=true, ZIndex=51 })
    MakeCorner(Notif, 14)
    MakeStroke(Notif, accentCol, 1, 0.45)

    New("Frame", { Parent=Notif, BackgroundColor3=accentCol,
        Size=UDim2.new(0,3,1,0), ZIndex=52 })

    local inner = New("Frame", {
        Parent=Notif, BackgroundTransparency=1,
        Position=UDim2.new(0,16,0,0), Size=UDim2.new(1,-20,0,0),
        AutomaticSize=Enum.AutomaticSize.Y, ZIndex=53 })
    MakePadding(inner, 12, 12, 0, 0)
    MakeList(inner, 3)

    New("TextLabel", { Parent=inner, BackgroundTransparency=1,
        Size=UDim2.new(1,0,0,17), Text=titleTxt, TextColor3=theme.Text,
        TextSize=13, Font=Enum.Font.GothamBold,
        TextXAlignment=Enum.TextXAlignment.Left, ZIndex=54 })
    New("TextLabel", { Parent=inner, BackgroundTransparency=1,
        Size=UDim2.new(1,0,0,0), AutomaticSize=Enum.AutomaticSize.Y,
        Text=bodyTxt, TextColor3=theme.SubText, TextSize=11,
        Font=Enum.Font.Gotham, TextXAlignment=Enum.TextXAlignment.Left,
        TextWrapped=true, ZIndex=54 })

    local ProgBg = New("Frame", { Parent=Notif, BackgroundColor3=theme.Border,
        AnchorPoint=Vector2.new(0,1), Position=UDim2.new(0,0,1,0),
        Size=UDim2.new(1,0,0,2), ZIndex=52 })
    local ProgFill = New("Frame", { Parent=ProgBg, BackgroundColor3=accentCol,
        Size=UDim2.new(1,0,1,0), ZIndex=53 })

    Notif.Position = UDim2.new(1,20,0,0)
    Tween(Notif,     { Position=UDim2.new(0,0,0,0) }, 0.35, Enum.EasingStyle.Back)
    Tween(ProgFill,  { Size=UDim2.new(0,0,1,0) }, dur, Enum.EasingStyle.Linear)
    task.delay(dur, function()
        Tween(Notif, { Position=UDim2.new(1,20,0,0) }, 0.28, Enum.EasingStyle.Quart, Enum.EasingDirection.In)
        task.wait(0.32)
        Notif:Destroy()
        NotifActive = math.max(0, NotifActive - 1)
    end)
end

local function ShowKeySystem(opts, theme, screenGui, onSuccess)
    local title      = opts.Title    or "Key Required"
    local subtitle   = opts.SubTitle or "Enter your access key to continue"
    local validKeys  = opts.Key      or {}
    local saveKey    = opts.SaveKey  == true
    local grabURL    = opts.GrabKey  or ""
    local keyPath    = "Mixusfield/__key.txt"

    if saveKey then
        local ok, valid = pcall(function()
            if isfile and isfile(keyPath) then
                local stored = readfile(keyPath)
                for _, k in ipairs(validKeys) do if stored == k then return true end end
            end
            return false
        end)
        if ok and valid then onSuccess(); return end
    end

    local Overlay = New("Frame", { Parent=screenGui, BackgroundColor3=Color3.fromRGB(0,0,0),
        BackgroundTransparency=0.45, Size=UDim2.new(1,0,1,0), ZIndex=200 })

    local function Close()
        Tween(Overlay, { BackgroundTransparency=1 }, 0.3)
        task.delay(0.35, function() if Overlay and Overlay.Parent then Overlay:Destroy() end end)
    end

    local Panel = New("Frame", { Parent=Overlay, BackgroundColor3=theme.KeyBg or theme.Background,
        AnchorPoint=Vector2.new(0.5,0.5), Position=UDim2.new(0.5,0,0.65,0),
        Size=UDim2.new(0,330,0,0), AutomaticSize=Enum.AutomaticSize.Y,
        ZIndex=201, ClipsDescendants=true })
    MakeCorner(Panel, 22)

-- DRAGGING  (fixed: uses absolute pos, no jump)
-- ════════════════════════════════════════════════════════════════
local function MakeDraggable(frame, handle)
    handle = handle or frame
    local dragging, dragStart, frameStart = false, nil, nil
    local velocity, lastCursor, lastTime = Vector2.zero, Vector2.zero, 0

    local function Clamp(px, py)
        local vp = ScreenSize()
        return math.clamp(px, 0, math.max(vp.X - frame.AbsoluteSize.X, 0)),
               math.clamp(py, 0, math.max(vp.Y - frame.AbsoluteSize.Y, 0))
    end

    handle.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1
        or input.UserInputType == Enum.UserInputType.Touch then
            dragging   = true
            dragStart  = Vector2.new(input.Position.X, input.Position.Y)
            frameStart = Vector2.new(frame.AbsolutePosition.X, frame.AbsolutePosition.Y)
            velocity   = Vector2.zero; lastCursor = dragStart; lastTime = tick()
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then dragging = false end
            end)
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement
        or input.UserInputType == Enum.UserInputType.Touch) then
            local now = tick(); local dt = math.max(now - lastTime, 0.001)
            local cur = Vector2.new(input.Position.X, input.Position.Y)
            velocity = (cur - lastCursor) / dt; lastCursor = cur; lastTime = now
            local cx, cy = Clamp(frameStart.X + (cur - dragStart).X, frameStart.Y + (cur - dragStart).Y)
            frame.Position = UDim2.new(0, cx, 0, cy)
        end
    end)

    UserInputService.InputEnded:Connect(function(input)
        if (input.UserInputType == Enum.UserInputType.MouseButton1
        or  input.UserInputType == Enum.UserInputType.Touch) and dragging then
            dragging = false
            local vx, vy, steps = velocity.X * 0.1, velocity.Y * 0.1, 14
            task.spawn(function()
                for i = 1, steps do
                    task.wait(1/60)
                    local decay = 1 - (i / steps)
                    local cx, cy = Clamp(frame.Position.X.Offset + vx*decay, frame.Position.Y.Offset + vy*decay)
                    frame.Position = UDim2.new(0, cx, 0, cy)
                end
            end)
        end
    end)
end

    -- ── SetTheme ─────────────────────────────────────────────────
    function WindowObj:SetTheme(name)
        local newT = Themes[name]; if not newT then return end
        self._Save:Set("__Theme", name)
        local dur = 0.3

        for _, obj in ipairs(ScreenGui:GetDescendants()) do
            local role = obj:GetAttribute("ThemeRole")
            local prop = obj:GetAttribute("ThemeColorProp") or "BackgroundColor3"
            if role and newT[role] then
                pcall(function() Tween(obj, {[prop]=newT[role]}, dur) end)
            end
            if obj:IsA("ScrollingFrame") then obj.ScrollBarImageColor3 = newT.ScrollBar end
        end
        for k,v in pairs(newT) do T[k]=v end

        for _, btnFr in ipairs(_topBtnFrames) do
            Tween(btnFr, {BackgroundColor3=newT.CardAlt}, dur)
        end
        for _, tabRef in ipairs(self._Tabs) do
            local isActive = tabRef._Title == self._ActiveTab
            Tween(tabRef._Line, {Size=UDim2.new(0, isActive and 22 or 0, 0,2.5),
                BackgroundColor3=isActive and newT.TabActive or newT.TabActive}, 0.25)
            if tabRef._IconBg  then Tween(tabRef._IconBg,  {BackgroundTransparency=isActive and 0.72 or 1}, 0.2) end
            if tabRef._TitleLbl then Tween(tabRef._TitleLbl,{TextColor3=isActive and newT.TabActive or newT.TabInactive},0.2) end
            if tabRef._IconImg then
                local col = isActive and newT.TabActive or newT.TabInactive
                if tabRef._IconImg:IsA("ImageLabel") then
                    Tween(tabRef._IconImg, {ImageColor3=col}, 0.2)
                else
                    Tween(tabRef._IconImg, {BackgroundColor3=col}, 0.2)
                end
            end
        end
    end

end
