local function Tween(obj, props, duration, style, direction)
    style     = style     or Enum.EasingStyle.Quart
    direction = direction or Enum.EasingDirection.Out
    local info = TweenInfo.new(duration or 0.25, style, direction)
    local t = TweenService:Create(obj, info, props)
    t:Play()
    return t
end

local function SafeCall(f, ...)
    if type(f) == "function" then
        local ok, err = pcall(f, ...)
        if not ok then warn("[NovaField] Callback error: " .. tostring(err)) end
    end
end

local function New(class, props, children)
    local obj = Instance.new(class)
    for k, v in pairs(props or {}) do
        if k ~= "Parent" then
            pcall(function() obj[k] = v end)
        end
    end
    for _, child in pairs(children or {}) do
        child.Parent = obj
    end
    if props and props.Parent then obj.Parent = props.Parent end
    return obj
end

local function Tag(obj, role)
    pcall(function() obj:SetAttribute("ThemeRole", role) end)
    return obj
end

local function MakePadding(parent, t, b, l, r)
    return New("UIPadding", {
        Parent        = parent,
        PaddingTop    = UDim.new(0, t or 0),
        PaddingBottom = UDim.new(0, b or 0),
        PaddingLeft   = UDim.new(0, l or 0),
        PaddingRight  = UDim.new(0, r or 0),
    })
end

local function MakeList(parent, padding, dir, halign)
    return New("UIListLayout", {
        Parent              = parent,
        Padding             = UDim.new(0, padding or 6),
        FillDirection       = dir or Enum.FillDirection.Vertical,
        HorizontalAlignment = halign or Enum.HorizontalAlignment.Left,
        SortOrder           = Enum.SortOrder.LayoutOrder,
    })
end

local function MakeCorner(parent, radius)
    return New("UICorner", { Parent = parent, CornerRadius = UDim.new(0, radius or 10) })
end

local function MakeStroke(parent, color, thickness, transparency)
    return New("UIStroke", {
        Parent       = parent,
        Color        = color or Color3.fromRGB(60, 20, 20),
        Thickness    = thickness or 1,
        Transparency = transparency or 0,
    })
end

local function IsMobile()
    return UserInputService.TouchEnabled and not UserInputService.KeyboardEnabled
end

local function ScreenSize()
    return Camera.ViewportSize
end

-- SAVE SYSTEM
-- ════════════════════════════════════════════════════════════════
local SaveSystem = {}
SaveSystem.__index = SaveSystem

function SaveSystem.new(configName)
    local self = setmetatable({}, SaveSystem)
    self.ConfigName = configName or "NovaFieldConfig"
    self.Data = {}
    self:Load()
    return self
end

function SaveSystem:Load()
    pcall(function()
        if not isfolder then return end
        if not isfolder("NovaField") then makefolder("NovaField") end
        local path = "NovaField/" .. self.ConfigName .. ".json"
        if isfile and isfile(path) then
            local raw = readfile(path)
            self.Data = HttpService:JSONDecode(raw) or {}
        end
    end)
end

function SaveSystem:Save()
    pcall(function()
        if not writefile then return end
        if makefolder and not isfolder("NovaField") then makefolder("NovaField") end
        local path = "NovaField/" .. self.ConfigName .. ".json"
        writefile(path, HttpService:JSONEncode(self.Data))
    end)
end

function SaveSystem:Set(key, value)
    self.Data[key] = value
    self:Save()
end

function SaveSystem:Get(key, default)
    if self.Data[key] ~= nil then return self.Data[key] end
    return default
end

-- FIX: DRAGGING SYSTEM — proper shadow sync, mobile-safe
-- ════════════════════════════════════════════════════════════════
local function MakeDraggable(frame, handle, onMoved)
    handle = handle or frame
    local dragging  = false
    local dragInput = nil
    local dragStart = nil
    local startPos  = nil
    local velocity  = Vector2.new(0, 0)
    local lastPos   = Vector2.new(0, 0)
    local lastTime  = 0

    local function ClampPos(px, py)
        local vp = ScreenSize()
        local fw = frame.AbsoluteSize.X
        local fh = frame.AbsoluteSize.Y
        return math.clamp(px, 0, vp.X - fw), math.clamp(py, 0, vp.Y - fh)
    end

    local function SetPos(cx, cy)
        frame.Position = UDim2.new(0, cx, 0, cy)
        if onMoved then onMoved(cx, cy) end  -- FIX: callback for shadow sync
    end

    handle.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1
        or input.UserInputType == Enum.UserInputType.Touch then
            dragging  = true
            dragStart = input.Position
            startPos  = frame.Position
            velocity  = Vector2.new(0, 0)
            lastPos   = Vector2.new(input.Position.X, input.Position.Y)
            lastTime  = tick()
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                end
            end)
        end
    end)

    handle.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement
        or input.UserInputType == Enum.UserInputType.Touch then
            dragInput = input
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if input == dragInput and dragging then
            local now    = tick()
            local dt     = math.max(now - lastTime, 0.001)
            local curPos = Vector2.new(input.Position.X, input.Position.Y)
            velocity = (curPos - lastPos) / dt
            lastPos  = curPos
            lastTime = now
            local delta = input.Position - dragStart
            local newX  = startPos.X.Offset + delta.X
            local newY  = startPos.Y.Offset + delta.Y
            local cx, cy = ClampPos(newX, newY)
            SetPos(cx, cy)
        end
    end)

    -- Inertia on release
    UserInputService.InputEnded:Connect(function(input)
        if (input.UserInputType == Enum.UserInputType.MouseButton1
        or input.UserInputType == Enum.UserInputType.Touch) and dragging then
            dragging = false
            local vx, vy = velocity.X * 0.1, velocity.Y * 0.1
            local steps  = 14
            task.spawn(function()
                for i = 1, steps do
                    task.wait(1/60)
                    local decay = 1 - (i / steps)
                    local cx, cy = ClampPos(
                        frame.Position.X.Offset + vx * decay,
                        frame.Position.Y.Offset + vy * decay
                    )
                    SetPos(cx, cy)
                end
            end)
        end
    end)
end

    -- FIX: SetTheme — applies to ALL tagged elements including strokes/toggles
    function WindowObj:SetTheme(name)
        local newT = Themes[name]
        if not newT then return end
        self._Save:Set("__Theme", name)
        local dur = 0.3
        for _, obj in ipairs(ScreenGui:GetDescendants()) do
            local ok, role = pcall(function() return obj:GetAttribute("ThemeRole") end)
            if ok and role and newT[role] then
                if obj:IsA("Frame") or obj:IsA("ScrollingFrame") then
                    pcall(function() Tween(obj, { BackgroundColor3 = newT[role] }, dur) end)
                elseif obj:IsA("TextLabel") or obj:IsA("TextBox") then
                    local textRoles = { Text=true, SubText=true, Accent=true, Muted=true, TabActive=true, TabInactive=true, SectionBar=true }
                    if textRoles[role] then
                        pcall(function() Tween(obj, { TextColor3 = newT[role] }, dur) end)
                    end
                elseif obj:IsA("ImageLabel") or obj:IsA("ImageButton") then
                    pcall(function() Tween(obj, { ImageColor3 = newT[role] }, dur) end)
                elseif obj:IsA("UIStroke") then
                    pcall(function() Tween(obj, { Color = newT[role] }, dur) end)
                end
            end
            if obj:IsA("ScrollingFrame") then
                pcall(function() obj.ScrollBarImageColor3 = newT.ScrollBar end)
            end
        end
        -- Update the current T table
        for k, v in pairs(newT) do T[k] = v end
        -- Update tab indicator colors
        for _, tabRef in ipairs(self._Tabs) do
            if tabRef._Title == self._ActiveTab then
                Tween(tabRef._Line,   { Size = UDim2.new(0, 22, 0, 2.5), BackgroundColor3 = newT.TabActive }, 0.25)
                if tabRef._IconLbl then Tween(tabRef._IconLbl, { TextColor3 = newT.TabActive }, 0.2) end
                if tabRef._IconBg  then Tween(tabRef._IconBg,  { BackgroundTransparency = 0.75 }, 0.2) end
            else
                Tween(tabRef._Line, { Size = UDim2.new(0, 0, 0, 2.5) }, 0.2)
                if tabRef._IconLbl then Tween(tabRef._IconLbl, { TextColor3 = newT.TabInactive }, 0.2) end
                if tabRef._IconBg  then Tween(tabRef._IconBg,  { BackgroundTransparency = 1 }, 0.2) end
            end
        end
    end

                local sliderDragging = false

                local function UpdateSlider(absX)
                    local trackPos  = SliderTrack.AbsolutePosition.X
                    local trackSize = SliderTrack.AbsoluteSize.X
                    local pct = math.clamp((absX - trackPos) / math.max(trackSize, 1), 0, 1)
                    local newVal = math.round(min + pct * (max - min))
                    if newVal == value then return end
                    value = newVal
                    ValLabel.Text   = tostring(value) .. suffix
                    SliderFill.Size = UDim2.new(pct, 0, 1, 0)
                    Knob.Position   = UDim2.new(pct, 0, 0.5, 0)
                    self._Save:Set(saveKey2, value)
                    SafeCall(callback, value)
                end

                SliderBtn.InputBegan:Connect(function(input)
                    if input.UserInputType == Enum.UserInputType.MouseButton1
                    or input.UserInputType == Enum.UserInputType.Touch then
                        sliderDragging = true
                        Tween(Knob, { Size = UDim2.new(0, 25, 0, 25) }, 0.1, Enum.EasingStyle.Back)
                        UpdateSlider(input.Position.X)
                    end
                end)
                UserInputService.InputChanged:Connect(function(input)
                    if sliderDragging and (input.UserInputType == Enum.UserInputType.MouseMovement
                    or input.UserInputType == Enum.UserInputType.Touch) then
                        UpdateSlider(input.Position.X)
                    end
                end)
                UserInputService.InputEnded:Connect(function(input)
                    if (input.UserInputType == Enum.UserInputType.MouseButton1
                    or input.UserInputType == Enum.UserInputType.Touch) and sliderDragging then
                        sliderDragging = false
                        Tween(Knob, { Size = UDim2.new(0, 20, 0, 20) }, 0.15)
                    end
                end)

                local el = { _Frame = Card, _Title = title }
                table.insert(self._Window._AllElements, el)

                local SliderObj = {}
                function SliderObj:Set(v)
                    value = math.clamp(v, min, max)

                    ValLabel.Text = tostring(value) .. suffix
                    Tween(SliderFill, { Size = UDim2.new(pct, 0, 1, 0) }, 0.2)
                    Tween(Knob,       { Position = UDim2.new(pct, 0, 0.5, 0) }, 0.2)
                    self._Save:Set(saveKey2, value); SafeCall(callback, value)
                end
                function SliderObj:Get() return value end
                return SliderObj
            end

                    Tag(StatCard, "Card")
                    MakeCorner(StatCard, 10)
                    MakeStroke(StatCard, T.Border, 1, 0)

                    New("TextLabel", {
                        Parent                 = StatCard,
                        BackgroundTransparency = 1,
                        AnchorPoint            = Vector2.new(0.5, 0),
                        Position               = UDim2.new(0.5, 0, 0, 9),
                        Size                   = UDim2.new(1, 0, 0, 14),
                        Text                   = stat.Label or "STAT",
                        TextColor3             = T.Muted,
                        TextSize               = 10,
                        Font                   = Enum.Font.GothamBold,
                        TextXAlignment         = Enum.TextXAlignment.Center,
                        ZIndex                 = 6,
                    })
                    New("TextLabel", {
                        Parent                 = StatCard,
                        BackgroundTransparency = 1,
                        AnchorPoint            = Vector2.new(0.5, 1),
                        Position               = UDim2.new(0.5, 0, 1, -9),
                        Size                   = UDim2.new(1, 0, 0, 20),
                        Text                   = stat.Value or "-",
                        TextColor3             = T.Accent,
                        TextSize               = 15,
                        Font                   = Enum.Font.GothamBold,
                        TextXAlignment         = Enum.TextXAlignment.Center,
                        ZIndex                 = 6,
                    })
                end
            end

            -- ── CreateActionRow ────────────────────────────────
            function SectionObj:CreateActionRow(opts)
                local rowTitle = opts.Title       or "Action"
                local rowDesc  = opts.Description or ""
                local callback = opts.Callback

                local Card = New("Frame", {
                    Parent           = SecContent,
                    BackgroundColor3 = T.Card,
                    Size             = UDim2.new(1, 0, 0, 52),
                    ZIndex           = 4,
                    LayoutOrder      = #SecContent:GetChildren(),
                })
                Tag(Card, "Card")
                MakeCorner(Card, 12)
                MakeStroke(Card, T.Border, 1, 0)
                MakePadding(Card, 0, 0, 15, 0)

                New("TextLabel", {
                    Parent                 = Card,
                    BackgroundTransparency = 1,
                    AnchorPoint            = Vector2.new(0, 0),
                    Position               = UDim2.new(0, 0, 0, 10),
                    Size                   = UDim2.new(0.8, 0, 0, 18),
                    Text                   = rowTitle,
                    TextColor3             = T.Text,
                    TextSize               = 14,
                    Font                   = Enum.Font.GothamBold,
                    TextXAlignment         = Enum.TextXAlignment.Left,
                    ZIndex                 = 5,
                })
                if rowDesc ~= "" then
                    New("TextLabel", {
                        Parent                 = Card,
                        BackgroundTransparency = 1,
                        AnchorPoint            = Vector2.new(0, 0),
                        Position               = UDim2.new(0, 0, 0, 30),
                        Size                   = UDim2.new(0.8, 0, 0, 14),
                        Text                   = rowDesc,
                        TextColor3             = T.SubText,
                        TextSize               = 11,
                        Font                   = Enum.Font.Gotham,
                        TextXAlignment         = Enum.TextXAlignment.Left,
                        ZIndex                 = 5,
                    })
                end
                New("TextLabel", {
                    Parent                 = Card,
                    BackgroundTransparency = 1,
                    AnchorPoint            = Vector2.new(1, 0.5),
                    Position               = UDim2.new(1, -10, 0.5, 0),
                    Size                   = UDim2.new(0, 18, 0, 18),
                    Text                   = ">",
                    TextColor3             = T.Muted,
                    TextSize               = 15,
                    Font                   = Enum.Font.GothamBold,
                    ZIndex                 = 5,
                })

                local ActionBtn = New("TextButton", {
                    Parent                 = Card,
                    BackgroundTransparency = 1,
                    Size                   = UDim2.new(1, 0, 1, 0),
                    Text                   = "",
                    ZIndex                 = 6,
                })
                ActionBtn.MouseEnter:Connect(function() Tween(Card, { BackgroundColor3 = T.CardAlt }, 0.15) end)
                ActionBtn.MouseLeave:Connect(function() Tween(Card, { BackgroundColor3 = T.Card }, 0.15) end)
                ActionBtn.MouseButton1Click:Connect(function() SafeCall(callback) end)
            end

            return SectionObj
        end -- CreateSection

        return TabObj
    end -- CreateTab

    -- ── Startup & Key ───────────────────────────────────────────
    if DoStartup then
        ShowStartupPopup({
            Title       = Title,
            OwnedBy     = Author,
            Description = SubTitle,
            Duration    = 3.5,
        }, T, ScreenGui)
    end

    if DoKey then
        local delay = DoStartup and 3.8 or 0.1
        task.delay(delay, function()
            ShowKeySystem(KeyOptions, T, ScreenGui, function()
                Window.Visible = true
                Shadow.Visible = true
                Window.Size = UDim2.new(0, WINDOW_W, 0, 0)
                task.delay(0.05, function()
                    Tween(Window, { Size = UDim2.new(0, WINDOW_W, 0, WINDOW_H) }, 0.4, Enum.EasingStyle.Back)
                end)
            end)
        end)
    else
        Window.Size = UDim2.new(0, WINDOW_W, 0, 0)
        task.delay(DoStartup and 3.8 or 0.1, function()
            Tween(Window, { Size = UDim2.new(0, WINDOW_W, 0, WINDOW_H) }, 0.4, Enum.EasingStyle.Back)
        end)
    end

    return WindowObj
end
