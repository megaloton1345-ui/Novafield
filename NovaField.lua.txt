local function Tween(obj, props, duration, style, direction)
    style     = style     or Enum.EasingStyle.Quart
    direction = direction or Enum.EasingDirection.Out
    local info = TweenInfo.new(duration or 0.25, style, direction)
    local t = TweenService:Create(obj, info, props)
    t:Play()
    return t
end

local function SafeCall(f, ...)
    if type(f) == "function" then
        local ok, err = pcall(f, ...)
        if not ok then warn("[NovaField] Callback error: " .. tostring(err)) end
    end
end

local function New(class, props, children)
    local obj = Instance.new(class)
    for k, v in pairs(props or {}) do
        if k ~= "Parent" then
            pcall(function() obj[k] = v end)
        end
    end
    for _, child in pairs(children or {}) do
        child.Parent = obj
    end
    if props and props.Parent then obj.Parent = props.Parent end
    return obj
end

local function Tag(obj, role)
    pcall(function() obj:SetAttribute("ThemeRole", role) end)
    return obj
end

local function MakePadding(parent, t, b, l, r)
    return New("UIPadding", {
        Parent        = parent,
        PaddingTop    = UDim.new(0, t or 0),
        PaddingBottom = UDim.new(0, b or 0),
        PaddingLeft   = UDim.new(0, l or 0),
        PaddingRight  = UDim.new(0, r or 0),
    })
end

local function MakeList(parent, padding, dir, halign)
    return New("UIListLayout", {
        Parent              = parent,
        Padding             = UDim.new(0, padding or 6),
        FillDirection       = dir or Enum.FillDirection.Vertical,
        HorizontalAlignment = halign or Enum.HorizontalAlignment.Left,
        SortOrder           = Enum.SortOrder.LayoutOrder,
    })
end

local function MakeCorner(parent, radius)
    return New("UICorner", { Parent = parent, CornerRadius = UDim.new(0, radius or 10) })
end

local function MakeStroke(parent, color, thickness, transparency)
    return New("UIStroke", {
        Parent       = parent,
        Color        = color or Color3.fromRGB(60, 20, 20),
        Thickness    = thickness or 1,
        Transparency = transparency or 0,
    })
end

local function IsMobile()
    return UserInputService.TouchEnabled and not UserInputService.KeyboardEnabled
end

local function ScreenSize()
    return Camera.ViewportSize
end

-- SAVE SYSTEM
-- ════════════════════════════════════════════════════════════════
local SaveSystem = {}
SaveSystem.__index = SaveSystem

function SaveSystem.new(configName)
    local self = setmetatable({}, SaveSystem)
    self.ConfigName = configName or "NovaFieldConfig"
    self.Data = {}
    self:Load()
    return self
end

function SaveSystem:Load()
    pcall(function()
        if not isfolder then return end
        if not isfolder("NovaField") then makefolder("NovaField") end
        local path = "NovaField/" .. self.ConfigName .. ".json"
        if isfile and isfile(path) then
            local raw = readfile(path)
            self.Data = HttpService:JSONDecode(raw) or {}
        end
    end)
end

function SaveSystem:Save()
    pcall(function()
        if not writefile then return end
        if makefolder and not isfolder("NovaField") then makefolder("NovaField") end
        local path = "NovaField/" .. self.ConfigName .. ".json"
        writefile(path, HttpService:JSONEncode(self.Data))
    end)
end

function SaveSystem:Set(key, value)
    self.Data[key] = value
    self:Save()
end

function SaveSystem:Get(key, default)
    if self.Data[key] ~= nil then return self.Data[key] end
    return default
end

local NotifActive = 0
local MAX_NOTIFS  = 4

local function ShowNotification(options, theme, screenGui)
    if NotifActive >= MAX_NOTIFS then return end
    NotifActive = NotifActive + 1

    local T      = options.Title    or "Notification"
    local C      = options.Content  or ""
    local dur    = options.Duration or 3
    local nType  = options.Type     or "Info"

    local typeColors = {
        Info    = theme.Accent,
        Success = Color3.fromRGB(28, 200, 115),
        Warning = Color3.fromRGB(255, 175, 28),
        Error   = Color3.fromRGB(255, 55, 55),
    }
    local accentColor = typeColors[nType] or theme.Accent

    local NotifHolder = screenGui:FindFirstChild("NovaNotifHolder")
    if not NotifHolder then
        NotifHolder = New("Frame", {
            Name                   = "NovaNotifHolder",
            Parent                 = screenGui,
            BackgroundTransparency = 1,
            AnchorPoint            = Vector2.new(1, 1),
            Position               = UDim2.new(1, -14, 1, -14),
            Size                   = UDim2.new(0, 295, 1, -14),
        })
        New("UIListLayout", {
            Parent            = NotifHolder,
            Padding           = UDim.new(0, 8),
            FillDirection     = Enum.FillDirection.Vertical,
            VerticalAlignment = Enum.VerticalAlignment.Bottom,
            HorizontalAlignment = Enum.HorizontalAlignment.Left,
            SortOrder         = Enum.SortOrder.LayoutOrder,

    end

        Size                   = UDim2.new(1, 0, 0, 0),

        Parent           = Notif,
        BackgroundColor3 = accentColor,
        Size             = UDim2.new(0, 3, 1, 0),

-- ════════════════════════════════════════════════════════════════
-- DRAGGING SYSTEM (smooth inertia + viewport clamping)
-- ════════════════════════════════════════════════════════════════
local function MakeDraggable(frame, handle)
    handle = handle or frame
    local dragging  = false
    local dragInput = nil
    local dragStart = nil
    local startPos  = nil
    local velocity  = Vector2.new(0, 0)
    local lastPos   = Vector2.new(0, 0)
    local lastTime  = 0

    local function ClampPos(px, py)
        local vp = ScreenSize()
        local fw = frame.AbsoluteSize.X
        local fh = frame.AbsoluteSize.Y
        return math.clamp(px, 0, vp.X - fw), math.clamp(py, 0, vp.Y - fh)
    end

    handle.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1
        or input.UserInputType == Enum.UserInputType.Touch then
            dragging  = true
            dragStart = input.Position
            startPos  = frame.Position
            velocity  = Vector2.new(0, 0)
            lastPos   = Vector2.new(input.Position.X, input.Position.Y)
            lastTime  = tick()
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                end
            end)
        end
    end)

    handle.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement
        or input.UserInputType == Enum.UserInputType.Touch then
            dragInput = input
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if input == dragInput and dragging then
            local now    = tick()
            local dt     = math.max(now - lastTime, 0.001)
            local curPos = Vector2.new(input.Position.X, input.Position.Y)
            velocity = (curPos - lastPos) / dt
            lastPos  = curPos
            lastTime = now
            local delta = input.Position - dragStart
            local newX  = startPos.X.Offset + delta.X
            local newY  = startPos.Y.Offset + delta.Y
            local cx, cy = ClampPos(newX, newY)
            frame.Position = UDim2.new(0, cx, 0, cy)
        end
    end)

    -- Inertia on release
    UserInputService.InputEnded:Connect(function(input)
        if (input.UserInputType == Enum.UserInputType.MouseButton1
        or input.UserInputType == Enum.UserInputType.Touch) and dragging then
            dragging = false
            local vx, vy = velocity.X * 0.1, velocity.Y * 0.1
            local steps  = 14
            task.spawn(function()
                for i = 1, steps do
                    task.wait(1/60)
                    local decay = 1 - (i / steps)
                    local cx, cy = ClampPos(
                        frame.Position.X.Offset + vx * decay,
                        frame.Position.Y.Offset + vy * decay
                    )
                    frame.Position = UDim2.new(0, cx, 0, cy)
                end
            end)
        end
    end)
end

    function WindowObj:SetTheme(name)
        local newT = Themes[name]
        if not newT then return end
        self._Save:Set("__Theme", name)
        local dur = 0.28
        for _, obj in ipairs(ScreenGui:GetDescendants()) do
            local ok, role = pcall(function() return obj:GetAttribute("ThemeRole") end)
            if ok and role and newT[role] then
                if obj:IsA("Frame") or obj:IsA("ScrollingFrame") then
                    Tween(obj, { BackgroundColor3 = newT[role] }, dur)
                elseif obj:IsA("TextLabel") or obj:IsA("TextBox") then
                    -- Only tween text roles (Text, SubText, Accent, Muted)
                    local textRoles = { Text=true, SubText=true, Accent=true, Muted=true }
                    if textRoles[role] then
                        Tween(obj, { TextColor3 = newT[role] }, dur)
                    end
                elseif obj:IsA("ImageLabel") or obj:IsA("ImageButton") then
                    Tween(obj, { ImageColor3 = newT[role] }, dur)
                elseif obj:IsA("UIStroke") then
                    Tween(obj, { Color = newT[role] }, dur)
                end
            end
            if obj:IsA("ScrollingFrame") then
                obj.ScrollBarImageColor3 = newT.ScrollBar
            end
        end
        for k, v in pairs(newT) do T[k] = v end
        for _, tabRef in ipairs(self._Tabs) do
            if tabRef._Title == self._ActiveTab then
                Tween(tabRef._Line,   { Size = UDim2.new(0, 22, 0, 2.5), BackgroundColor3 = newT.TabActive }, 0.25)
                if tabRef._IconLbl then Tween(tabRef._IconLbl, { TextColor3 = newT.TabActive }, 0.2) end
                if tabRef._IconBg  then Tween(tabRef._IconBg,  { BackgroundTransparency = 0.75 }, 0.2) end
            else
                Tween(tabRef._Line, { Size = UDim2.new(0, 0, 0, 2.5) }, 0.2)
                if tabRef._IconLbl then Tween(tabRef._IconLbl, { TextColor3 = newT.TabInactive }, 0.2) end
                if tabRef._IconBg  then Tween(tabRef._IconBg,  { BackgroundTransparency = 1 }, 0.2) end
            end
        end
    end

                    local trackPos  = SliderTrack.AbsolutePosition.X

                    local pct = math.clamp((absX - trackPos) / math.max(trackSize, 1), 0, 1)
                    local newVal = math.round(min + pct * (max - min))
                    if newVal == value then return end
                    value = newVal
                    ValLabel.Text   = tostring(value) .. suffix

                end

                return SliderObj
            end

                    TextXAlignment         = Enum.TextXAlignment.Center,
                    ZIndex                 = 5,
                })

                local Btn = New("TextButton", {
                    Parent                 = Card,
                    BackgroundTransparency = 1,
                    Size                   = UDim2.new(1, 0, 1, 0),
                    Text                   = "",
                    ZIndex                 = 6,
                })
                Btn.MouseEnter:Connect(function() Tween(Card, { BackgroundColor3 = T.CardAlt }, 0.15) end)
                Btn.MouseLeave:Connect(function() Tween(Card, { BackgroundColor3 = T.Card }, 0.15) end)
                Btn.MouseButton1Click:Connect(function() SafeCall(callback) end)

                local el = { _Frame = Card, _Title = label }
                table.insert(self._Window._AllElements, el)
            end

            -- ── CreateKeybind ──────────────────────────────────
            function SectionObj:CreateKeybind(opts)
                local title    = opts.Title   or "Keybind"
                local default  = opts.Default or Enum.KeyCode.RightShift
                local saveKey2 = opts.Flag    or title
                local callback = opts.Callback

                local Card = MakeCard(#SecContent:GetChildren())

                New("TextLabel", {
                    Parent                 = Card,
                    BackgroundTransparency = 1,
                    Size                   = UDim2.new(0.6, 0, 0, 18),
                    Text                   = title,
                    TextColor3             = T.Text,
                    TextSize               = 14,
                    Font                   = Enum.Font.GothamBold,
                    TextXAlignment         = Enum.TextXAlignment.Left,
                    ZIndex                 = 5,
                })

                local KeyBadge = New("Frame", {
                    Parent           = Card,
                    BackgroundColor3 = T.CardAlt,
                    AnchorPoint      = Vector2.new(1, 0.5),
                    Position         = UDim2.new(1, 0, 0.5, 0),
                    Size             = UDim2.new(0, 95, 0, 30),
                    ZIndex           = 5,
                })
                MakeCorner(KeyBadge, 9)
                MakeStroke(KeyBadge, T.Border, 1, 0)

                local KeyLabel = New("TextLabel", {
                    Parent                 = KeyBadge,
                    BackgroundTransparency = 1,
                    Size                   = UDim2.new(1, 0, 1, 0),
                    Text                   = tostring(default.Name),
                    TextColor3             = T.Accent,
                    TextSize               = 12,
                    Font                   = Enum.Font.GothamBold,
                    TextXAlignment         = Enum.TextXAlignment.Center,
                    ZIndex                 = 6,
                })

                local binding = false
                local key     = default

                local KeyBtn = New("TextButton", {
                    Parent                 = KeyBadge,
                    BackgroundTransparency = 1,
                    Size                   = UDim2.new(1, 0, 1, 0),
                    Text                   = "",
                    ZIndex                 = 7,
                })

                KeyBtn.MouseButton1Click:Connect(function()
                    binding = true
                    KeyLabel.Text = "..."
                    Tween(KeyBadge, { BackgroundColor3 = T.AccentDark }, 0.15)
                end)

                UserInputService.InputBegan:Connect(function(input, processed)
                    if binding and input.UserInputType == Enum.UserInputType.Keyboard then
                        key = input.KeyCode
                        KeyLabel.Text = input.KeyCode.Name
                        binding = false
                        Tween(KeyBadge, { BackgroundColor3 = T.CardAlt }, 0.15)
                        self._Save:Set(saveKey2, input.KeyCode.Name)
                        return
                    end
                    if not processed and input.KeyCode == key then
                        SafeCall(callback, key)
                    end
                end)

                local el = { _Frame = Card, _Title = title }
                table.insert(self._Window._AllElements, el)
            end

            -- ── CreateColorPicker ──────────────────────────────
            function SectionObj:CreateColorPicker(opts)
                local title    = opts.Title   or "Color"
                local default  = opts.Default or Color3.fromRGB(230, 60, 75)
                local saveKey2 = opts.Flag    or title
                local callback = opts.Callback

                local Card = MakeCard(#SecContent:GetChildren())

                New("TextLabel", {
                    Parent                 = Card,
                    BackgroundTransparency = 1,
                    Size                   = UDim2.new(1, 0, 0, 18),
                    Text                   = title,
                    TextColor3             = T.Text,
                    TextSize               = 14,
                    Font                   = Enum.Font.GothamBold,
                    TextXAlignment         = Enum.TextXAlignment.Left,
                    ZIndex                 = 5,
                })

                local presets = {
                    Color3.fromRGB(230, 60, 75),
                    Color3.fromRGB(255, 110, 30),
                    Color3.fromRGB(255, 205, 30),
                    Color3.fromRGB(28, 200, 115),
                    Color3.fromRGB(30, 155, 255),
                    Color3.fromRGB(165, 96, 255),
                    Color3.fromRGB(255, 255, 255),
                    Color3.fromRGB(90, 90, 90),
                }

                local SwatchRow = New("Frame", {
                    Parent                 = Card,
                    BackgroundTransparency = 1,
                    Size                   = UDim2.new(1, 0, 0, 28),
                    ZIndex                 = 5,
                })
                New("UIListLayout", {
                    Parent        = SwatchRow,
                    FillDirection = Enum.FillDirection.Horizontal,
                    Padding       = UDim.new(0, 6),
                    SortOrder     = Enum.SortOrder.LayoutOrder,
                })

                local selectedColor = default

                -- Preview bar
                local Preview = New("Frame", {
                    Parent           = Card,
                    BackgroundColor3 = default,
                    Size             = UDim2.new(1, 0, 0, 22),
                    ZIndex           = 5,
                })
                MakeCorner(Preview, 7)
                MakeStroke(Preview, T.Border, 1, 0)

                for _, c in ipairs(presets) do
                    local sw = New("Frame", {
                        Parent           = SwatchRow,
                        BackgroundColor3 = c,
                        Size             = UDim2.new(0, 24, 0, 24),
                        ZIndex           = 6,
                    })
                    MakeCorner(sw, 6)
                    local swBtn = New("TextButton", {
                        Parent                 = sw,
                        BackgroundTransparency = 1,
                        Size                   = UDim2.new(1, 0, 1, 0),
                        Text                   = "",
                        ZIndex                 = 7,
                    })
                    swBtn.MouseButton1Click:Connect(function()
                        selectedColor = c
                        Preview.BackgroundColor3 = c
                        SafeCall(callback, c)
                    end)
                    swBtn.MouseButton1Down:Connect(function()
                        Tween(sw, { Size = UDim2.new(0, 20, 0, 20) }, 0.08)
                    end)
                    swBtn.MouseButton1Up:Connect(function()
                        Tween(sw, { Size = UDim2.new(0, 24, 0, 24) }, 0.13, Enum.EasingStyle.Back)
                    end)
                end

                local el = { _Frame = Card, _Title = title }
                table.insert(self._Window._AllElements, el)

                local ColorObj = {}
                function ColorObj:Set(c) selectedColor = c; Preview.BackgroundColor3 = c; SafeCall(callback, c) end
                function ColorObj:Get() return selectedColor end
                return ColorObj
            end

            return SectionObj
        end -- CreateSection

        return TabObj
    end -- CreateTab

    -- ── Startup & Key ───────────────────────────────────────────
    if DoStartup then
        ShowStartupPopup({
            Title       = Title,
            OwnedBy     = Author,
            Description = SubTitle,
            Duration    = 3.5,
        }, T, ScreenGui)
    end

    if DoKey then
        local delay = DoStartup and 3.8 or 0.1
        task.delay(delay, function()
            ShowKeySystem(KeyOptions, T, ScreenGui, function()
                Window.Visible = true
                Shadow.Visible = true
                Window.Size = UDim2.new(0, WINDOW_W, 0, 0)
                task.delay(0.05, function()
                    Tween(Window, { Size = UDim2.new(0, WINDOW_W, 0, WINDOW_H) }, 0.4, Enum.EasingStyle.Back)
                end)
            end)
        end)
    else
        Window.Size = UDim2.new(0, WINDOW_W, 0, 0)
        task.delay(DoStartup and 3.8 or 0.1, function()
            Tween(Window, { Size = UDim2.new(0, WINDOW_W, 0, WINDOW_H) }, 0.4, Enum.EasingStyle.Back)
        end)
    end

    return WindowObj
end
